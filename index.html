function ambilSinyalTerakhirDariSheet(dari, untuk, jenis, callback) { panggilAPI("ambilPesanPribadi", [dari, untuk]).then(dataPesan => { if (!dataPesan || !Array.isArray(dataPesan)) return; // Ambil waktu saat ini untuk validasi sinyal baru const waktuSekarang = new Date().getTime(); for (let i = dataPesan.length - 1; i >= 0; i--) { const msg = dataPesan[i]; if (msg.pengirim.toLowerCase() === dari.toLowerCase() && msg.penerima.toLowerCase() === untuk.toLowerCase()) { if (msg.teks.startsWith(`WEBRTC_SIGNAL:${jenis}|`)) { try { // VALIDASI KHUSUS UNTUK SINYAL "END" if (jenis === "END" && msg.timestamp) { // Jika sinyal END dibuat lebih dari 15 detik yang lalu, abaikan (itu riwayat lama) if (waktuSekarang - msg.timestamp > 15000) { continue; } } const dataMentahJson = msg.teks.split(`WEBRTC_SIGNAL:${jenis}|`)[1]; callback(JSON.parse(dataMentahJson)); break; } catch (e) { console.error("Gagal parse sinyal:", e); } } } } }); }
